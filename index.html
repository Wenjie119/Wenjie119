def find_paradox_cases(predictions, ground_truths, 
                        geo_types, threshold=0.4):
    paradoxes = {'high_cc_low_gfs': [], 'low_cc_high_gfs': []}
    for v_pred, v_true, geo in zip(predictions, ground_truths, geo_types):
        cc  = pearsonr(v_pred.flat, v_true.flat)[0]

        elif gfs - cc > threshold:    # Low CC but high GFS
            paradoxes['low_cc_high_gfs'].append(
                (v_pred, v_true, cc, gfs, geo))
    # Sort by largest discrepancy
    for k in paradoxes:
        paradoxes[k].sort(key=lambda x: abs(x[2]-x[3]), reverse=True)
    return paradoxes

\subsection{Geological Interface Segmentation}
Given velocity model $\mathbf{v}\in\mathbb{R}^{N_x\times N_z}$, 
we compute the gradient magnitude field:
\begin{equation}
G_{ij} = \sqrt{(\partial_x v)_{ij}^2 + (\partial_z v)_{ij}^2}
\end{equation}
Geological interfaces (layer boundaries, fault planes, 
salt flanks) are identified as high-gradient regions:
\begin{equation}
S_{ij}(\mathbf{v};\tau) = \mathbb{1}[G_{ij} > \tau]
\end{equation}
The threshold $\tau = 50$ m/s per grid point is 
determined by maximizing $F_1$ score on the 
GeoFWI validation set (ROC analysis over 
$\tau \in [20, 200]$ m/s with step 10 m/s; 
peak $F_1 = \textbf{[TBD]}$ at $\tau{=}\textbf{[TBD]}$ m/s).

\subsection{Geological Fidelity Score}
\begin{equation}
\boxed{
\mathrm{GFS}(\mathbf{v}_{\mathrm{pred}}, \mathbf{v}_{\mathrm{true}}) 
= \tfrac{1}{2}\Big[
\underbrace{\mathrm{IoU}(S_{\mathrm{pred}}, S_{\mathrm{true}})}_{\text{structural overlap}}
+
\underbrace{\tfrac{1}{2}\big(1 + \overline{\cos\angle(\nabla\mathbf{v}_{\mathrm{pred}}, \nabla\mathbf{v}_{\mathrm{true}})}\big)}_{\text{gradient alignment in }[0,1]}
\Big]
\in [0,1]
}
\end{equation}
where the inner product and norms are averaged over interface pixels 
$\Omega = \{(i,j): S_{ij}(\mathbf{v}_{\mathrm{true}})=1\}$.

"""
gfs.py — Geological Fidelity Score
Full implementation for GeoFWI Paper 2
"""
import numpy as np
from scipy.ndimage import sobel
from sklearn.metrics import f1_score


def gradient_magnitude(v: np.ndarray) -> np.ndarray:
    """Compute velocity gradient magnitude using Sobel operator."""
    gx = sobel(v, axis=1)   # horizontal gradient
    gz = sobel(v, axis=0)   # vertical gradient
    return np.sqrt(gx**2 + gz**2)


def segment_interfaces(v: np.ndarray, tau: float = 50.0) -> np.ndarray:
    """
    Binary segmentation of geological interfaces.
    
    Args:
        v:   velocity model, shape (Nz, Nx), units m/s
        tau: gradient threshold in m/s per grid point
    
    Returns:
        Binary interface map (1 = interface, 0 = background)
    """
    return (gradient_magnitude(v) > tau).astype(float)


def iou_binary(s_pred: np.ndarray, s_true: np.ndarray) -> float:
    """Intersection over Union for binary segmentation maps."""
    inter = np.logical_and(s_pred, s_true).sum()
    union = np.logical_or(s_pred,  s_true).sum()
    return float(inter) / (float(union) + 1e-8)


def gradient_direction_consistency(
    v_pred: np.ndarray,
    v_true: np.ndarray,
    s_true: np.ndarray
) -> float:
    """
    Mean cosine similarity of velocity gradients on true interface pixels,
    rescaled to [0, 1] via (cos_sim + 1) / 2 so GFS stays in [0, 1].
    """
    mask = s_true.astype(bool)
    if mask.sum() == 0:
        return 1.0

    gx_p = sobel(v_pred, axis=1)
    gz_p = sobel(v_pred, axis=0)
    gx_t = sobel(v_true, axis=1)
    gz_t = sobel(v_true, axis=0)

    dot    = gx_p[mask] * gx_t[mask] + gz_p[mask] * gz_t[mask]
    norm_p = np.sqrt(gx_p[mask]**2 + gz_p[mask]**2) + 1e-8
    norm_t = np.sqrt(gx_t[mask]**2 + gz_t[mask]**2) + 1e-8
    cos_sim = dot / (norm_p * norm_t)

    return float(np.mean((cos_sim + 1.0) * 0.5))


def compute_gfs(
    v_pred: np.ndarray,
    v_true: np.ndarray,
    tau: float = 50.0
) -> float:
    """
    Geological Fidelity Score.
    
    Args:
        v_pred: predicted velocity model (Nz, Nx), m/s
        v_true: ground-truth velocity model (Nz, Nx), m/s  
        tau:    interface detection threshold, m/s/grid
    
    Returns:
        GFS in [0, 1]. Higher = better geological fidelity.
    """
    s_pred = segment_interfaces(v_pred, tau)
    s_true = segment_interfaces(v_true, tau)
    
    iou = iou_binary(s_pred, s_true)
    gdc = gradient_direction_consistency(v_pred, v_true, s_true)

    return 0.5 * (iou + gdc)  # gdc already in [0,1] via (cos+1)/2


# ── ROC analysis to find optimal τ ──────────────────────────────
def find_optimal_tau(
    val_predictions: list,
    val_ground_truths: list,
    tau_range: np.ndarray = np.arange(20, 205, 5),
) -> dict:
    """
    Find optimal threshold τ by maximizing F1 on validation set.
    
    Returns dict with 'optimal_tau', 'f1_scores', 'tau_range'.
    """
    f1_scores = []
    for tau in tau_range:
        f1_batch = []
        for v_pred, v_true in zip(val_predictions, val_ground_truths):
            s_pred = segment_interfaces(v_pred, tau).flatten()
            s_true = segment_interfaces(v_true, tau).flatten()
            # Use v_true segmentation as ground truth label
            f1_batch.append(f1_score(s_true, s_pred, zero_division=0))
        f1_scores.append(np.mean(f1_batch))
    
    best_idx = np.argmax(f1_scores)
    return {
        'optimal_tau': tau_range[best_idx],
        'best_f1':     f1_scores[best_idx],
        'f1_scores':   f1_scores,
        'tau_range':   tau_range
    }


# ── Batch evaluation ──────────────────────────────────────────────
def evaluate_all_metrics(
    predictions:   list,
    ground_truths: list,
    tau: float = 50.0
) -> dict:
    """
    Compute CC, SSIM, RMSE, GFS for a batch of predictions.
    Returns dict with lists of per-sample metrics.
    """
    from scipy.stats import pearsonr
    from skimage.metrics import structural_similarity as ssim
    
    results = {'cc': [], 'ssim': [], 'rmse': [], 'gfs': []}
    
    for v_pred, v_true in zip(predictions, ground_truths):
        cc, _  = pearsonr(v_pred.flatten(), v_true.flatten())
        s_score = ssim(v_pred, v_true,
                       data_range=v_true.max()-v_true.min())
        rmse   = np.sqrt(np.mean((v_pred - v_true)**2)) / 1000.  # km/s
        gfs    = compute_gfs(v_pred, v_true, tau)
        
        results['cc'].append(cc)
        results['ssim'].append(s_score)
        results['rmse'].append(rmse)
        results['gfs'].append(gfs)
    
    return {k: np.array(v) for k, v in results.items()}


# ── Kendall τ stratified analysis ─────────────────────────────────
def kendall_tau_by_geo_type(
    metrics:    dict,
    geo_labels: np.ndarray
) -> dict:
    """Compute Kendall τ(GFS, CC) and τ(GFS, SSIM) per geological type."""
    from scipy.stats import kendalltau
    
    results = {}
    for geo_type in np.unique(geo_labels):
        mask = geo_labels == geo_type
        tau_cc,   _ = kendalltau(metrics['gfs'][mask], 
                                   metrics['cc'][mask])
        tau_ssim, _ = kendalltau(metrics['gfs'][mask], 
                                   metrics['ssim'][mask])
        tau_rmse, _ = kendalltau(metrics['gfs'][mask], 
                                   metrics['rmse'][mask])
        results[geo_type] = {
            'n':         mask.sum(),
            'tau_cc':    round(tau_cc,   3),
            'tau_ssim':  round(tau_ssim, 3),
            'tau_rmse':  round(tau_rmse, 3),
        }
    return results

SEG/EAGE 2D Salt Model — the hardest test
▾

    

      
Source: https://wiki.seg.org/wiki/SEG/EAGE_Salt_and_Overthrust_Models

      
Why it's hard: Very high velocity contrast between salt (4480 m/s) and surrounding sediments (1500–2500 m/s). This is exactly the scenario where GFS should shine and CC will mislead.

      
Expected result: Zero-shot CC may be low (salt body is hard to localize from GeoFWI training), but GFS should improve significantly with fine-tuning even from N=10 samples, because proximal guidance constrains physically impossible velocity jumps.

    

  


  

    
"""
transfer.py — Few-shot transfer learning for Paper 3
GeoFWI pretrained model → classical FWI benchmarks
"""
import torch
import torch.nn as nn
from copy import deepcopy
import numpy as np

def few_shot_finetune(
    pretrained_model,
    target_dataset,
    n_shot:      int   = 50,
    epochs:      int   = 50,
    lr:          float = 1e-5,
    seed:        int   = 42,
    freeze_body: bool  = True,
) -> nn.Module:
    """
    Fine-tune a GeoFWI-pretrained model on n_shot target samples.
    
    If freeze_body=True, only the last 2 decoder layers are updated.
    This is the recommended setting for n_shot < 100.
    """
    torch.manual_seed(seed)
    model = deepcopy(pretrained_model)
    model.train()
    
    if freeze_body:
        for param in model.parameters():
            param.requires_grad = False
        # Unfreeze decoder tail (last 2 blocks)
        for param in model.decoder[-2:].parameters():
            param.requires_grad = True
        for param in model.projection_head.parameters():
            param.requires_grad = True
    
    # Sample n_shot examples
    idx = np.random.choice(len(target_dataset), 
                           min(n_shot, len(target_dataset)), 
                           replace=False)
    subset = torch.utils.data.Subset(target_dataset, idx)
    loader = torch.utils.data.DataLoader(
        subset, batch_size=min(8, n_shot), shuffle=True
    )
    
    optimizer = torch.optim.AdamW(
        filter(lambda p: p.requires_grad, model.parameters()),
        lr=lr, weight_decay=1e-4
    )
    
    for epoch in range(epochs):
        for seismic, v_true in loader:
            seismic = seismic.cuda()
            v_true  = v_true.cuda()
            
            v_pred = model(seismic)           # forward pass
            loss   = nn.MSELoss()(v_pred, v_true)  # reconstruction loss
            # Note: proximal guidance is applied inside model.forward()
            # so no extra physics term needed here
            
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
    
    model.eval()
    return model


def run_transfer_experiment(
    pretrained_ckpt: str,
    target_datasets: dict,         # {'Marmousi2': DataLoader, ...}
    shot_counts:    list = [0, 10, 50, 100, 500],
    n_seeds:        int  = 3,
) -> dict:
    """Run full transfer experiment matrix. Returns results dict."""
    model = load_gcfwi(pretrained_ckpt)
    results = {}
    
    for dataset_name, (train_ds, test_loader) in target_datasets.items():
        results[dataset_name] = {}
        for n in shot_counts:
            cc_runs = []
            for seed in range(n_seeds):
                if n == 0:
                    ft_model = model  # zero-shot: no fine-tuning
                else:
                    ft_model = few_shot_finetune(
                        model, train_ds, n_shot=n, seed=seed
                    )
                cc = evaluate_cc(ft_model, test_loader)
                cc_runs.append(cc)
            
            results[dataset_name][n] = {
                'mean': np.mean(cc_runs),
                'std':  np.std(cc_runs)
            }
            print(f"{dataset_name} N={n:4d}: "
                  f"CC={np.mean(cc_runs):.3f}±{np.std(cc_runs):.3f}")
    
    return results
